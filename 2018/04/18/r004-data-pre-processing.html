Order

sort(iris$Sepal.Length)
rev(iris$Sepal.Length)

Calculation

iris$Sepal2 = with(iris,iris$Sepal.Length/iris$Sepal.Width)

Delet

rm(iris)

Missing value

Detect missing value

y <- c(1,2,3,NA) 
is.na(y)

data(sleep, package= "VIM") 
sleep[!complete.cases(sleep),]

#### Number of missing values
!complete.cases(sleep) 
sum(!complete.cases(sleep))

#### Missing value ratio 
mean(complete.cases(sleep))

Delet missing value

sleep <- na.omit(sleep)

Replace missing value: Multiple Imputation

##### 3types: miss completely at random MCAR, missing at random MAR, missing not at random MNAR
library(mice)
data(sleep, package='VIM')
imp <- mice(sleep, seed=100)
fit <- with(imp,lm(Dream ~ Span + Gest))
pooled <- pool(fit)
summary(pooled)

Change data

Modifying variable type

is.numeric(a) 
is.logical(a)
a <- as.character(a)

day <-c("01/01/2011")
day <- as.Date(day,'%m/%d/%Y')

Rename old variable

fix(iris)

names(iris)[1] <- 'sl'

library("reshape") ##### but indeed, dplyr is more useful
iris <- rename(iris, c(Sepal.Length="sl"))
detach("package:reshape", unload=TRUE)

Variable coding

iris$sl <- cut(iris$Sepal.Length,c(min(iris$Sepal.Length),5.1,6.4,max(iris$Sepal.Length)))
levels(iris$sl) <- c('short','medium','long')

iris <- within(iris,{
        sl <- NA
        sl[iris$Sepal.Length>=6.4] <- 'long'
        sl[iris$Sepal.Length>=5.1&iris$Sepal.Length<6.4] <- 'medium'
        sl[iris$Sepal.Length<5.1] <- 'short'
        })

4.4 Data pre-processing: dplyr

library(dplyr)

summary()

tbl_df(iris)
glimpse(iris)
summarise(iris,meanswidth=mean(Sepal.Width,na.rm = TRUE))

subset

Subset rows: filter

iris[iris$Sepal.Length==5 & iris$Petal.Length==1.6,]
filter(iris,Sepal.Length==5, Petal.Length==1.6)
filter(iris,Sepal.Length==5 | Petal.Length==1.6)
filter(iris,Sepal.Length>7 , Petal.Length<6)

slice(iris, 9:15)

Subset columns

select(iris, Sepal.Width, Petal.Length, Species)
select(iris,Sepal.Width:Species)

distinct(select(iris,Petal.Length))

Order colums

arrange(iris,Sepal.Width,Sepal.Length) ##### Ascending order, exsist priority
arrange(iris,desc(Sepal.Width)) ##### Descending order

Sample

sample_frac(iris, 0.99, replace = TRUE)
glimpse(distinct(glimpse(sample_frac(iris, 0.99, replace = TRUE))))
sample_n(iris, 10, replace = TRUE)

Change data

Rename old variable

rename(data,newname=oldname)

generate new variable

newiris <- mutate(iris,Sepal2=Sepal.Length*Sepal.Width,Sepal0=Sepal.Length/Sepal.Width)

group the data

iris_byspecies <- group_by(iris,Species)
slmean <- summarise(iris_byspecies, count = n(),
                   slmean = mean(Sepal.Length, na.rm = TRUE),
                   plmean = mean(Sepal.Length, na.rm = TRUE))
slmean <- filter(slmean, count > 20, plmean < 6)
