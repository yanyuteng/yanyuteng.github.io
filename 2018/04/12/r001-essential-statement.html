Loop statement

while loop

Fibonacci<-NULL 
Fibonacci[1]<-Fibonacci[2]<-1
i=1
while (Fibonacci[i]+Fibonacci[i+1]<1000){
    Fibonacci[i+2]<-Fibonacci[i]+Fibonacci[i+1] 
    i = i+1
    } 
Fibonacci

for loop

Fibonacci<-NULL
Fibonacci[1]<-Fibonacci[2]<-1
n=16
for (i in 3:n) {
    Fibonacci[i]<-Fibonacci[i-2]+Fibonacci[i-1]
    }
Fibonacci

repeat loop

Fibonacci<-NULL
Fibonacci[1]<-Fibonacci[2]<-1 
i<-1
repeat { 
        Fibonacci[i+2]<-Fibonacci[i]+Fibonacci[i+1] 
        i<-i+1
        if (Fibonacci[i]+Fibonacci[i+1]>=1000) 
        break
    }

mean & var

var.mean=function(x){
        i=x
        m=mean(i)
        v=var(i) 
        return(list(mean=m,var=v))
        } 
y=c(5,15,32,25,26,28,65,48,3,37,45,54,23,44) 
var.mean(y)

x=rnorm(100,mean=60,sd=1)

        }
