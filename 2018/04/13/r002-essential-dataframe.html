R002 Essential dataframe

Posted on 2018-04-13   |   In Quantitative Method   |   This article has been read times

Vector

generate vector

a <- c(1,4,7,3,7)
b <- c('one','two','three')
c <- c(TRUE,FALSE,TRUE)
d <- seq(3,20,by=2) 
e <- seq(3,20,length=9)

read vector

a[1:3]
a[c(1,3)]

### read the vector's order; R from 1, but Python from 0
which.min(a)
which.max(a)
which(a==7)

### read the vector's number
max(a)
a[a<7]

### judge the vector's type
mode(a)
is.vector(a)
is.function(c)

delete vector

a[-2]
a[-2:-4]

change vector

rev(a)
sort(a)
paste(a,e)

Matrix

generate matrix

a <- c(1:12)
a <- matrix(a,nrow = 3,ncol = 4,byrow = T)

read matrix

a[2,]
a[,2]
a[2,2]
a[2,c(2,3)]
a[1:2,2]

change matrix’s names

a <- matrix(1:9,nrow = 3,ncol = 3,byrow = T)
rnames = c('num','height','weight')
cnames = c('cat','dog','pig')
rownames(a) = rnames
colnames(a) = cnames

rownames(a) = c('num','height','weight')
colnames(a) = c('cat','dog','pig')

change victor to matrix

a = c(1:6)
dim(a) = c(2,3)

computer

set.seed(100)
a = matrix(sample(100,15),nrow = 5,ncol = 3)
set.seed(101)
b = matrix(sample(100,15),nrow = 5,ncol = 3)

basic computer

a*2
a^2
a-2
colSums(a)
colMeans(a)[1]
rowMeans(a)
rowMeans(a)

matrix*matrix (linear algebra)

#### front rows*latter cols
t(a) %*% b 

#### t(a) %*% b = crossprod(a,b)
#### t(a)*a = crossprod(a)

diagonal matrix

diag(c(1,2,3,4))
diag(a)
diag(3)
diag(diag(a))

solve the linear simultaneous equations

#### inverse matrix
a = matrix(rnorm(16),4,4)
solve(a) 

#### ax=b
a = matrix(rnorm(16),4,4)
b = c(1:4)
solve(a,b)

#### consider xa=b?
is it the t(a)t(x)=t(b) ?
a = matrix(rnorm(16),4,4)
b = matrix(1:4,nrow = 1)
solve(t(a),t(b))

eigenvalu

a = diag(4) + 1
a.e = eigen(a,symmetric = T)

Data.frame (two-axis matrix)

generate

a1 = c(1,2,3)
a2 = c(2,3,4)
a = data.frame(a1,a2)
a = data.frame('weight'=a1,'price'=a2)

a = data.frame(a1='weight',a2='price')


patientID = c(1, 2, 3, 4)
age = c(25, 34, 28, 52)
diabetes = c("Type1", "Type2", "Type1", "Type1")
status = c("Poor", "Improved", "Excellent", "Poor") 
patientdata = data.frame(patientID, age, diabetes, status) 
patientdata

read

patientdata[,1:2]
patientdata[1:2,]

Array (Multidimensional Matrix)

dim1 = c("A1", "A2")
dim2 = c("B1", "B2", "B3")
dim3 = c("C1", "C2", "C3", "C4")
dim4 = c("D1", "D2", "D3")
z = array(1:72, c(2, 3, 4, 3), dimnames=list(dim1, dim2, dim3, dim4))
z[1,2,3,]

List

generate list

x = matrix(1:20, nrow=5, ncol=4, byrow=TRUE)
swim = read.csv("http://www.macalester.edu/~kaplan/ISM/datasets/swim100m.csv")
examplelist=list(swim,x)

read list

examplelist[[2]][1:3,2:4]

Factor

generate

examplefactor = factor(rep(c(1:3), times=5) )
a = sample(100,15)

mean, quantile, sum etc.

tapply(a,examplefactor,mean)

test tapply

rbind(a,examplefactor)
boo =rbind(a,examplefactor)[2,] == 2
which(boo)
rbind(a,examplefactor)[1,which(boo)]
sum(rbind(a,examplefactor)[1,which(boo)])/length(which(boo))
